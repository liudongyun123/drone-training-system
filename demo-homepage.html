滚动透明渐变效果 (HomePage.tsx)

// 动态透明度控制
const [heroOpacity, setHeroOpacity] = useState(1);

useEffect(() => {
  const handleScroll = () => {
    const scrolled = window.scrollY;
    const opacity = Math.max(0.7, 1 - scrolled / 500);
    setHeroOpacity(opacity);
  };
  
  window.addEventListener('scroll', handleScroll);
  return () => window.removeEventListener('scroll', handleScroll);
}, []);

// 在组件中使用
<section 
  className="relative overflow-hidden text-white pt-24 pb-32"
  style={{
    background: `linear-gradient(135deg, 
      rgba(37, 99, 235, ${heroOpacity}) 0%, 
      rgba(124, 58, 237, ${heroOpacity}) 50%, 
      rgba(14, 165, 233, ${heroOpacity}) 100%)`,
    transition: 'background 0.3s ease'
  }}
>
  {/* 页面内容 */}
</section>
